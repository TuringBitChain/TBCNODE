## 2. 改造目标（C-B + P-2）

```cpp
bool ConnectBlock(...) {
    AssertLockHeld(cs_main);
    CCoinsViewCache view(pcoinsTip.get());
    DoVerifyBlockScripts(block, view, ...);
    UpdateCoinsForBlock(block, view, pindex);

    // 三锁原子提交（v2.6.1 §5 锁次序）
    {
        std::unique_lock smtx_lock(g_mempool.smtx);
        std::unique_lock bw_lock(pcoinsTip->batchWriteMtx);

        // 1. C-B: UpdateTip(seqlock) 在 view.Flush() 之前
        //    worker 任何时刻拍 snap 都看到 "tip_hash 跟 script_flags 同 epoch"
        chainActive.SetTip(pindex);   // 老路径仍保留
        tbc::validation::g_chainstate.UpdateTip(
            pindex,
            GetBlockScriptFlags(config, pindex),
            config.GetGenesisActivationHeight(),
            IsGenesisEnabled(config, pindex)
        );

        // 2. C-B: Flush UTXO（cacheCoinsConcurrent 通过 P0.2 双写已同步；BatchWrite 持 unique）
        if (!view.Flush()) {
            // P-2: Flush 失败硬 abort
            //   seqlock 已写新 tip 但 cacheCoins 半新半旧 → 不可恢复
            //   abort + LevelDB 重启恢复（chainstate format 不变）
            LogPrintf("FATAL: view.Flush() failed after UpdateTip(%s); abort\n",
                      pindex->GetBlockHash().ToString());
            std::abort();
            // 不调析构 → 锁不释放，但进程退出由 OS 释放虚拟地址空间 + LevelDB LOCK 文件
        }

        // 3. 移除 mempool 中已上链 tx
        g_mempool.RemoveForBlockNL(block.vtx, ...);
    }   // 释放 smtx + bw

    // 4. 异步刷新 GBT 模板（不阻塞）
    g_gbt_snapshot.RefreshAsync();

    // 5. 异步信号分发
    g_signal_dispatcher.EnqueueBlock(SignalType::BlockConnected, pindex->GetBlockHash());

    return true;
}
```
