### 3.2 实现（thread_local 栈 + level 检查）

```cpp
// src/sync.cpp
#include "validation/lock_hierarchy.h"
#include <vector>
#include <cstdlib>

#ifdef DEBUG_LOCKORDER

namespace {
struct LockEntry {
    void* mutex_addr;
    bool is_recursive;
    int recursive_depth;
    int level;
    const char* file;
    int line;
    const char* name;
};

thread_local std::vector<LockEntry> g_lock_stack;
}

void EnterCritical(const char* pszName, const char* pszFile, int nLine,
                   void* cs, int level, bool fTry, bool isRecursive)
{
    // 1. 重入识别
    for (auto& e : g_lock_stack) {
        if (e.mutex_addr == cs && e.is_recursive) {
            ++e.recursive_depth;
            return;
        }
    }

    // 2. 层级检查（栈顶 level 必须 < 新锁 level）
    if (!g_lock_stack.empty()) {
        int top_level = g_lock_stack.back().level;
        if (level <= top_level && !fTry) {
            LogPrintf("LOCK ORDER VIOLATION:\n"
                      "  trying to lock %s (level %d) at %s:%d\n"
                      "  already held %s (level %d) at %s:%d\n"
                      "  full stack: %s\n",
                      pszName, level, pszFile, nLine,
                      g_lock_stack.back().name, top_level,
                      g_lock_stack.back().file, g_lock_stack.back().line,
                      LocksHeld().c_str());
            std::abort();
        }
    }

    // 3. 入栈
    g_lock_stack.push_back({cs, isRecursive, 1, level, pszFile, nLine, pszName});
}

void LeaveCritical(void* cs) {
    for (auto it = g_lock_stack.rbegin(); it != g_lock_stack.rend(); ++it) {
        if (it->mutex_addr == cs) {
            if (it->is_recursive && it->recursive_depth > 1) {
                --it->recursive_depth;
            } else {
                g_lock_stack.erase(std::next(it).base());
            }
            return;
        }
    }
    LogPrintf("LeaveCritical: mutex %p not in stack\n", cs);
    std::abort();
}

#endif
```

### 3.3 boost::recursive_mutex 重入识别

```cpp
// CCriticalSection（src/sync.h:90）
class CCriticalSection : public AnnotatedMixin<boost::recursive_mutex> {
public:
    static constexpr bool kIsRecursive = true;   // 用于 EnterCritical isRecursive 参数
};

// DebugLock 模板特化
template <typename Mutex>
class DebugLock {
    boost::unique_lock<Mutex> lock;
public:
    DebugLock(Mutex& m, const char* name, const char* file, int line,
              int level = LEVEL_DEFAULT, bool fTry = false)
    {
        EnterCritical(name, file, line, &m, level, fTry,
                      Mutex::kIsRecursive);   // SFINAE 默认 false
        if (fTry) lock = boost::unique_lock<Mutex>(m, boost::try_to_lock);
        else lock.lock();
    }
    ~DebugLock() { LeaveCritical(&lock.mutex()); }
};
```

## 5. 测试用例

### 5.1 单元测试

```cpp
// src/test/sync_tests.cpp
TEST(SyncHook, ReentrantLockSameThread) {
    CCriticalSection cs;
    LOCK(cs);
    LOCK(cs);   // 重入应该 OK
    SUCCEED();
}

TEST(SyncHook, LockOrderViolationAborts) {
    CCriticalSection a, b;
    LOCK_LVL(b, 1);   // 持低 level
    EXPECT_DEATH({ LOCK_LVL(a, 0); }, "LOCK ORDER VIOLATION");
}

TEST(SyncHook, DefaultLevelNoConflictWithExisting) {
    CCriticalSection a, b;   // 都未标 level，走 LEVEL_DEFAULT
    LOCK(a);
    LOCK(b);   // LEVEL_DEFAULT == LEVEL_DEFAULT，不应触发违反
    SUCCEED();
}

TEST(SyncHook, NewMutexWithLevelLowerThanDefault) {
    CCriticalSection cs_low;        // 假设标 LEVEL_CS_MAIN = 0
    CCriticalSection cs_unlabeled;  // LEVEL_DEFAULT
    LOCK_LVL(cs_low, 0);
    LOCK(cs_unlabeled);   // 0 → DEFAULT 严格递增，OK
    SUCCEED();
}
```
