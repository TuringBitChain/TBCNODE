## 3. 代码改造

```cpp
bool CCoinsViewCache::GetCoin(const COutPoint& outpoint, Coin& coin) const {
    // 老路径（非 worker hot path）：用 GetCoinConcurrent 内部实现
    return GetCoinConcurrent(outpoint, coin);
}

bool CCoinsViewCache::HaveCoin(const COutPoint& outpoint) const {
    Coin c;
    return GetCoinConcurrent(outpoint, c);
}

void CCoinsViewCache::AddCoin(const COutPoint& outpoint, Coin&& coin, bool overwrite, ...) {
    // 老路径仍用 std::unique_lock(batchWriteMtx) 保证互斥
    std::unique_lock bw(batchWriteMtx);
    size_t new_usage = memusage::DynamicUsage(coin);
    CCoinsCacheEntry entry;
    entry.coin = std::move(coin);
    entry.flags = CCoinsCacheEntry::DIRTY | (overwrite ? 0 : CCoinsCacheEntry::FRESH);
    bool ins = cacheCoins.insert(outpoint, std::move(entry));
    if (ins) cachedCoinsUsage.fetch_add(new_usage);
    else if (overwrite) {
        cacheCoins.update_fn(outpoint, [&](CCoinsCacheEntry& e) noexcept {
            cachedCoinsUsage.fetch_sub(memusage::DynamicUsage(e));
            e = std::move(entry);
            cachedCoinsUsage.fetch_add(new_usage);
        });
    }
}

bool CCoinsViewCache::SpendCoin(const COutPoint& outpoint, Coin* moveto) {
    std::unique_lock bw(batchWriteMtx);
    bool spent = false;
    cacheCoins.update_fn(outpoint, [&](CCoinsCacheEntry& e) noexcept {
        if (moveto) *moveto = e.coin;
        cachedCoinsUsage.fetch_sub(memusage::DynamicUsage(e));
        e.coin.Clear();
        e.flags |= CCoinsCacheEntry::DIRTY;
        spent = true;
    });
    return spent;
}
```
