### 3.1 smoke round-trip 测试

```python
# test/functional/binary_roundtrip_smoke.py
"""
P0.0b smoke round-trip：
  1. v1 binary 跑 1000 regtest 块，关停
  2. vN-smoke binary（含 P0.0a sync.h 改造 + P1 早期 Chainstate seqlock）启动同一 datadir
     期望：无 reindex，立即同步到 tip
  3. vN-smoke binary 关停
  4. v1 binary 启动同一 datadir，期望同样无 reindex

注意：vN-smoke 此时只有 sync.h hook + Chainstate seqlock，没有 cacheCoins 改造、没有 dispatcher。
真 round-trip 在 P5.6 P3 末 full 才测。
"""

from test_framework.test_framework import BitcoinTestFramework
import subprocess, os, time

class SmokeRoundtripTest(BitcoinTestFramework):
    def set_test_params(self):
        self.num_nodes = 1
        # 用 v1 binary 启动节点
        self.binary_v1 = os.environ.get("V1_BINARY", "build-v1/src/bitcoind")
        self.binary_vn = os.environ.get("VN_SMOKE_BINARY", "build/src/bitcoind")

    def run_test(self):
        # Step 1: v1 跑 1000 块
        self.stop_node(0)
        self.options.bitcoind = self.binary_v1
        self.start_node(0)
        self.nodes[0].generatetoaddress(1000, self.nodes[0].getnewaddress())
        v1_tip = self.nodes[0].getbestblockhash()
        self.stop_node(0)

        # Step 2: vN-smoke 启动同 datadir
        self.options.bitcoind = self.binary_vn
        t0 = time.time()
        self.start_node(0)
        startup_time = time.time() - t0
        assert startup_time < 30, f"Startup too slow ({startup_time}s), suggests reindex"
        vn_tip = self.nodes[0].getbestblockhash()
        assert v1_tip == vn_tip, f"Tip mismatch: v1={v1_tip} vs vn={vn_tip}"

        # Step 3: vN-smoke 跑 100 块再关停
        self.nodes[0].generatetoaddress(100, self.nodes[0].getnewaddress())
        vn_tip2 = self.nodes[0].getbestblockhash()
        self.stop_node(0)

        # Step 4: v1 启动同 datadir
        self.options.bitcoind = self.binary_v1
        t0 = time.time()
        self.start_node(0)
        startup_time = time.time() - t0
        assert startup_time < 30, f"Startup too slow ({startup_time}s)"
        v1_tip2 = self.nodes[0].getbestblockhash()
        assert v1_tip2 == vn_tip2, f"Reverse tip mismatch"

if __name__ == "__main__":
    SmokeRoundtripTest().main()
```
