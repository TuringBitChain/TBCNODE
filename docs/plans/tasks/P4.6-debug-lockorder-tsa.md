## 3. 代码改造

```cpp
// 例：src/coins.h
class CCoinsViewCache {
    libcuckoo::cuckoohash_map<...> cacheCoins;
    mutable std::shared_mutex batchWriteMtx;
    mutable std::shared_mutex metaMtx;

    uint256 hashBlock GUARDED_BY(metaMtx);
    std::atomic<size_t> cachedCoinsUsage{0};

public:
    bool BatchWrite(...) override
        EXCLUSIVE_LOCKS_REQUIRED(!batchWriteMtx) /* 内部取 unique */;

    bool GetCoinConcurrent(...) const
        SHARED_LOCKS_REQUIRED(!batchWriteMtx) /* 内部取 shared */;
};
```
