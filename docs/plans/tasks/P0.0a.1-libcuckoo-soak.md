### 3.2 soak 测试结构

```cpp
// src/test/libcuckoo_soak_tests.cpp
#include <libcuckoo/cuckoohash_map.hh>
#include <boost/test/unit_test.hpp>
#include <atomic>
#include <thread>
#include <chrono>

BOOST_AUTO_TEST_SUITE(libcuckoo_soak)

BOOST_AUTO_TEST_CASE(soak_50M_entries_24h) {
    libcuckoo::cuckoohash_map<uint64_t, std::array<uint8_t, 64>> map;
    constexpr size_t N = 50'000'000;
    constexpr int THREADS = 32;
    constexpr auto DURATION = std::chrono::hours(24);

    std::atomic<size_t> ops{0};
    std::atomic<bool> stop{false};

    // 预填 N 个 entry
    for (size_t i = 0; i < N; i++) {
        std::array<uint8_t, 64> v{};
        std::memcpy(v.data(), &i, sizeof(i));
        BOOST_REQUIRE(map.insert(i, v));
    }

    auto t_start = std::chrono::steady_clock::now();

    std::vector<std::thread> threads;
    for (int t = 0; t < THREADS; t++) {
        threads.emplace_back([&, t] {
            std::mt19937_64 rng(t);
            while (!stop.load()) {
                uint64_t k = rng() % N;
                std::array<uint8_t, 64> v;
                if (map.find(k, v)) {
                    ops.fetch_add(1, std::memory_order_relaxed);
                }
                if ((ops.load() & 0xFFFF) == 0) {
                    map.update_fn(k, [&](auto& val) { val[0]++; });
                }
            }
        });
    }

    std::this_thread::sleep_for(DURATION);
    stop.store(true);
    for (auto& th : threads) th.join();

    BOOST_TEST_MESSAGE("ops total: " << ops.load());
    BOOST_REQUIRE_GT(ops.load(), 1'000'000'000ULL);
}

BOOST_AUTO_TEST_SUITE_END()
```

### 3.3 内存监控脚本

```bash
# tools/soak_memory_watch.sh
PID=$1
DURATION_H=$2
INTERVAL_S=60
LOG=soak_memory_${PID}.log

START_RSS=$(ps -o rss= -p $PID)
echo "$(date +%s) start_rss=${START_RSS}KB" > $LOG

end_time=$(($(date +%s) + DURATION_H * 3600))
while [ $(date +%s) -lt $end_time ]; do
    RSS=$(ps -o rss= -p $PID)
    echo "$(date +%s) rss=${RSS}KB" >> $LOG
    sleep $INTERVAL_S
done

END_RSS=$(ps -o rss= -p $PID)
DRIFT_KB=$((END_RSS - START_RSS))
echo "DRIFT: ${DRIFT_KB}KB" >> $LOG
test $DRIFT_KB -lt 102400  # < 100MB
```

## 7. 回滚条件

任一 KPI 不达标 → libcuckoo 不集成；进备选方案：`std::shared_mutex` + `std::unordered_map`（性能损失 30-40%）。整体方案 v2.6.1 中 P0 卡片需要全部重写 → 触发 M-1a 失败 → **整套 v2.6.1 放弃**，沉没 5 周。
