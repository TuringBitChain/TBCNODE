## 1. 目标

把 `src/coins.h:329` 的 `mutable std::mutex mCoinsViewCacheMtx` + `std::unordered_map<COutPoint, CCoinsCacheEntry> cacheCoins` 替换成：

```cpp
libcuckoo::cuckoohash_map<COutPoint, CCoinsCacheEntry, SaltedOutpointHasher> cacheCoins;
mutable std::shared_mutex batchWriteMtx;       // LEVEL_BATCHWRITE_MTX = 2
mutable std::shared_mutex metaMtx;             // LEVEL_META_MTX = 3
```

不改任何方法签名（GetCoin/HaveCoin/AddCoin/SpendCoin/BatchWrite/Flush 全部保留），只改实现。本卡专注**数据结构替换 + 锁声明**，BatchWrite/GetCoinConcurrent 实现细节在 P0.2/P0.3 完成。

```cpp
// src/coins.h（改造后）
#include <libcuckoo/cuckoohash_map.hh>
#include <shared_mutex>
#include "validation/lock_hierarchy.h"

class CCoinsViewCache : public CCoinsViewBacked {
public:
    CCoinsViewCache(CCoinsView* baseIn);
    ~CCoinsViewCache();

    // 老接口签名 100% 保留
    bool GetCoin(const COutPoint &outpoint, Coin &coin) const override;
    bool HaveCoin(const COutPoint &outpoint) const override;
    void AddCoin(const COutPoint &outpoint, Coin &&coin, bool overwrite, ...);
    bool SpendCoin(const COutPoint &outpoint, Coin* moveto = nullptr);
    bool BatchWrite(CCoinsMap &mapCoins, const uint256 &hashBlock) override;
    bool Flush();
    uint256 GetBestBlock() const override;
    void SetBestBlock(const uint256 &hashBlock);
    size_t DynamicMemoryUsage() const;

    // 新接口（P0.3 实现）
    bool GetCoinConcurrent(const COutPoint &outpoint, Coin &coin) const;
    bool HaveCoinConcurrent(const COutPoint &outpoint) const;
    bool IsBatchWriteInProgress() const;

private:
    // 改造后数据结构
    libcuckoo::cuckoohash_map<COutPoint, CCoinsCacheEntry, SaltedOutpointHasher> cacheCoins;

    // 锁层级（详见 lock-hierarchy.md）
    mutable std::shared_mutex batchWriteMtx;       // level 2
    mutable std::shared_mutex metaMtx;             // level 3

    uint256 hashBlock;                             // GUARDED_BY(metaMtx)
    std::atomic<size_t> cachedCoinsUsage{0};       // P0.4b 实现

    // 二级 LRU（P0.3 实现）
    // mutable LRUCache<COutPoint, Coin> levelTwoCache;   // P0.3 加
};
```
